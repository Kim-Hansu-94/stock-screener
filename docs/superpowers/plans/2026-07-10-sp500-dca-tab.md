```python
from __future__ import annotations

from datetime import date, timedelta

import pandas as pd
import yfinance as yf

from . import prices_kr

SP500_TICKER = "^GSPC"
ETF_TICKER = "360750"
ETF_NAME = "TIGER 미국S&P500"
HIGH_LOOKBACK_DAYS = 252
# 표시 3년 + 52주 고점 계산 여유. 수신 첫 1년은 축소 윈도우 근사(참고 데이터라 허용).
LOOKBACK_DAYS = 1460


def compute_daily_state(close: pd.Series) -> pd.DataFrame:
    """지수 종가에서 52주(252거래일) 고점과 고점 대비 하락률(%)을 계산한다."""
    high_52w = close.rolling(HIGH_LOOKBACK_DAYS, min_periods=1).max()
    drawdown_pct = (close / high_52w - 1.0) * 100.0
    return pd.DataFrame({
        "close": close, "high_52w": high_52w, "drawdown_pct": drawdown_pct,
    })


def daily_rows(daily: pd.DataFrame) -> list[dict]:
    return [
        {
            "date": idx.date().isoformat(),
            "close": round(float(row["close"]), 2),
            "high_52w": round(float(row["high_52w"]), 2),
            "drawdown_pct": round(float(row["drawdown_pct"]), 2),
        }
        for idx, row in daily.iterrows()
    ]


def fetch_index_history(today: date) -> pd.Series:
    start = today - timedelta(days=LOOKBACK_DAYS)
    df = yf.download(SP500_TICKER, start=start.isoformat(), auto_adjust=True, progress=False)
    close = df["Close"]
    if isinstance(close, pd.DataFrame):
        close = close[SP500_TICKER]
    return close.dropna()


def etf_quote_row(hist: pd.DataFrame) -> dict:
    close = hist["Close"].dropna()
    last = float(close.iloc[-1])
    prev = float(close.iloc[-2]) if len(close) >= 2 else last
    idx = close.index[-1]
    return {
        "ticker": ETF_TICKER,
        "name": ETF_NAME,
        "close": round(last, 2),
        "change_pct": round((last / prev - 1.0) * 100.0, 2),
        "currency": "KRW",
        "as_of": (idx.date() if hasattr(idx, "date") else idx).isoformat(),
    }


def etf_history_rows(hist: pd.DataFrame) -> list[dict]:
    clean = hist.dropna(subset=["Open", "High", "Low", "Close", "Volume"])
    return [
        {
            "ticker": ETF_TICKER,
            "market": "KR",
            "date": idx.date().isoformat() if hasattr(idx, "date") else str(idx)[:10],
            "open": float(row["Open"]),
            "high": float(row["High"]),
            "low": float(row["Low"]),
            "close": float(row["Close"]),
            "volume": int(row["Volume"]),
        }
        for idx, row in clean.iterrows()
    ]


def run(db, today: date) -> None:
    """일일 파이프라인 진입점. sp500_daily는 비어 있으면 전체, 이후 최근 30일 창만 upsert."""
    close = fetch_index_history(today)
    daily = compute_daily_state(close)
    rows = daily_rows(daily)

    latest = db.get_sp500_latest_date()
    if latest:
        cutoff = (date.fromisoformat(latest) - timedelta(days=30)).isoformat()
        rows = [r for r in rows if r["date"] >= cutoff]
    db.save_sp500_daily(rows)

    # TIGER 시세·일봉은 표시용 — 실패해도 지수 데이터 저장에는 영향 없음
    try:
        hist = prices_kr.get_kr_stock_history(ETF_TICKER, today, LOOKBACK_DAYS)
        db.save_price_history(etf_history_rows(hist))
        db.save_sp500_etf_quotes([etf_quote_row(hist)])
    except Exception as exc:
        print(f"  TIGER 시세 수집 실패 (건너뜀): {exc}", flush=True)

    print(f"  sp500_daily {len(rows)}행, 기준일 {daily.index[-1].date()}", flush=True)
```

```typescript
// S&P 500 적립 통계 순수 함수. 데이터는 서버 컴포넌트가 넘겨준다.
export interface Purchase {
  id: number
  date: string
  shares: number
  price: number
}

export interface DcaStats {
  totalInvested: number
  totalShares: number
  avgPrice: number
  currentValue: number
  returnPct: number
}

export function computeDcaStats(purchases: Purchase[], lastClose: number): DcaStats | null {
  if (purchases.length === 0) return null
  const totalInvested = purchases.reduce((s, p) => s + p.shares * p.price, 0)
  const totalShares = purchases.reduce((s, p) => s + p.shares, 0)
  const currentValue = totalShares * lastClose
  return {
    totalInvested,
    totalShares,
    avgPrice: totalInvested / totalShares,
    currentValue,
    returnPct: (currentValue / totalInvested - 1) * 100,
  }
}

export function hasPurchaseInMonth(purchases: Purchase[], yearMonth: string): boolean {
  return purchases.some((p) => p.date.slice(0, 7) === yearMonth)
}

export function kstYearMonth(now: Date = new Date()): string {
  const kst = new Date(now.getTime() + 9 * 3600 * 1000)
  return kst.toISOString().slice(0, 7)
}
```

```typescript
'use server'

import { revalidateTag } from 'next/cache'
import { SP500_PURCHASES_TAG } from '@/lib/sp500'
import { createServerSupabaseClient } from '@/lib/supabase'

export interface ActionResult {
  ok: boolean
  error?: string
}

export async function addPurchase(formData: FormData): Promise<ActionResult> {
  const dateStr = String(formData.get('date') ?? '')
  const shares = Number(formData.get('shares'))
  const price = Number(formData.get('price'))

  if (!/^\d{4}-\d{2}-\d{2}$/.test(dateStr)) return { ok: false, error: '날짜 형식이 올바르지 않습니다.' }
  if (!Number.isFinite(shares) || shares <= 0) return { ok: false, error: '수량은 0보다 커야 합니다.' }
  if (!Number.isFinite(price) || price <= 0) return { ok: false, error: '단가는 0보다 커야 합니다.' }

  const supabase = createServerSupabaseClient()
  const { error } = await supabase
    .from('sp500_purchases')
    .insert({ date: dateStr, shares, price })
  if (error) return { ok: false, error: error.message }

  revalidateTag(SP500_PURCHASES_TAG)
  return { ok: true }
}

export async function deletePurchase(id: number): Promise<ActionResult> {
  if (!Number.isInteger(id) || id <= 0) return { ok: false, error: '잘못된 기록입니다.' }

  const supabase = createServerSupabaseClient()
  const { error } = await supabase.from('sp500_purchases').delete().eq('id', id)
  if (error) return { ok: false, error: error.message }

  revalidateTag(SP500_PURCHASES_TAG)
  return { ok: true }
}
```

TIGER 차트(매수 마커 포함)와 지수 참고 차트 두 곳에서 재사용한다. 마커 시각은 데이터에 존재하는 날짜여야 하므로, 매수일 이하의 가장 가까운 봉 날짜로 스냅한다.

```tsx
'use client'

import { useEffect, useRef } from 'react'
import { createChart, LineStyle } from 'lightweight-charts'

export interface ChartPoint {
  time: string
  value: number
}

export interface ChartMarkerInput {
  date: string
  text: string
}

interface Sp500ChartProps {
  data: ChartPoint[]
  markers?: ChartMarkerInput[]
  priceLines?: Array<{ price: number; title: string }>
  height?: number
}

// 마커 날짜를 그 이하의 가장 가까운 데이터 날짜로 스냅 (주말·휴장일 매수 입력 대비)
function snapMarkers(data: ChartPoint[], markers: ChartMarkerInput[]) {
  const times = data.map((d) => d.time)
  return markers
    .map((m) => {
      let snapped: string | null = null
      for (const t of times) {
        if (t <= m.date) snapped = t
        else break
      }
      return snapped ? { time: snapped, text: m.text } : null
    })
    .filter((m): m is { time: string; text: string } => m !== null)
}

export function Sp500Chart({ data, markers = [], priceLines = [], height = 320 }: Sp500ChartProps) {
  const containerRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    if (!containerRef.current || data.length === 0) return
    const chart = createChart(containerRef.current, {
      height,
      layout: { textColor: '#374151' },
      rightPriceScale: { borderVisible: false },
      timeScale: { borderVisible: false },
    })
    const series = chart.addLineSeries({ color: '#2563eb', lineWidth: 2 })
    series.setData(data)

    if (markers.length > 0) {
      series.setMarkers(
        snapMarkers(data, markers).map((m) => ({
          time: m.time,
          position: 'belowBar' as const,
          color: '#dc2626',
          shape: 'arrowUp' as const,
          text: m.text,
        })),
      )
    }
    for (const line of priceLines) {
      series.createPriceLine({
        price: line.price, color: '#6b7280', lineWidth: 1,
        lineStyle: LineStyle.Dashed, title: line.title,
      })
    }
    chart.timeScale().fitContent()

    const onResize = () => chart.applyOptions({ width: containerRef.current?.clientWidth ?? 600 })
    onResize()
    window.addEventListener('resize', onResize)
    return () => {
      window.removeEventListener('resize', onResize)
      chart.remove()
    }
  }, [data, markers, priceLines, height])

  return <div ref={containerRef} />
}
```
