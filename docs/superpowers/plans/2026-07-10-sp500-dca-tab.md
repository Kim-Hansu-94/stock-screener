**Goal:** S&P 500 조정 단계(52주 고점 대비 하락률)를 매일 계산·저장하고, `/sp500` 탭에서 현재 단계·매수 플랜·차트·이력을 보여준다.

## Global Constraints

- 단계 정의: 52주(252거래일) 최고 종가 대비 하락률. 1단계 -5%, 2단계 -10%, 3단계 -20%, 4단계 -30%. 경계값은 깊은 단계로 판정(정확히 -5.0%는 1단계).
- 예비자금 비율 기본값: 1단계 20% / 2단계 30% / 3단계 30% / 4단계 20% (누적 20/50/80/100).

- 지수 데이터: ^GSPC, 2010-01-01부터. UI 문구는 전부 한국어.
- pytest는 리포 루트에서 실행(`pytest pipeline/tests/...`), 임포트는 `from pipeline.src....`.

```python
"""
S&P 500 단계별 분할 매수 백테스트
==================================
매월 첫 거래일 $1,000 적립을 공통 전제로 세 전략 비교:
  A) 즉시 매수 (정기 DCA)
  B) 단계별 분할 — 조정 단계 진입 시 (사이클 투입액+현금)의 누적 목표
     비율(20/50/80/100%)까지 투입, 신고가 회복 시 사이클 리셋
  C) 단일 임계값 — 사이클당 -10% 첫 진입 시 현금 전량 투입
기간: 2000-01-01~현재, 2010-01-01~현재 두 구간
실행: python pipeline/research/backtest_sp500_stages.py
"""
from __future__ import annotations

from pathlib import Path

import pandas as pd
import yfinance as yf

OUT_CSV = Path(__file__).parent / "backtest_sp500_stages_summary.csv"
MONTHLY = 1_000.0
HIGH_LOOKBACK = 252
CUM_TARGET = {1: 0.20, 2: 0.50, 3: 0.80, 4: 1.00}
THRESHOLDS = [(4, -30.0), (3, -20.0), (2, -10.0), (1, -5.0)]


def stage_for(dd: float) -> int:
    for stage, th in THRESHOLDS:
        if dd <= th + 1e-9:
            return stage
    return 0


def load_daily(start: str) -> pd.DataFrame:
    raw = yf.download("^GSPC", start=start, auto_adjust=True, progress=False)
    close = raw["Close"]
    if isinstance(close, pd.DataFrame):
        close = close["^GSPC"]
    close = close.dropna()
    high = close.rolling(HIGH_LOOKBACK, min_periods=1).max()
    dd = (close / high - 1.0) * 100.0
    stages = dd.map(stage_for)
    cycle_ids, cycle_id, dipped = [], 0, False
    for d, s in zip(dd, stages):
        if dipped and d >= -1e-9:
            cycle_id += 1
            dipped = False
        if s >= 1:
            dipped = True
        cycle_ids.append(cycle_id)
    df = pd.DataFrame({"close": close, "dd": dd, "stage": stages, "cycle": cycle_ids})
    periods = df.index.to_period("M")
    df["is_contrib_day"] = ~periods.duplicated(keep="first")
    return df


def run_regular(df: pd.DataFrame) -> dict:
    shares = invested = 0.0
    for row in df.itertuples():
        if row.is_contrib_day:
            shares += MONTHLY / row.close
            invested += MONTHLY
    return _summary("A. 정기 DCA", df, shares, 0.0, invested)


def run_staged(df: pd.DataFrame) -> dict:
    cash = shares = invested = cycle_spent = 0.0
    seen_cycle, max_stage = -1, 0
    for row in df.itertuples():
        if row.is_contrib_day:
            cash += MONTHLY
            invested += MONTHLY
        if row.cycle != seen_cycle:
            seen_cycle, max_stage, cycle_spent = row.cycle, 0, 0.0
        if row.stage > max_stage:
            target = CUM_TARGET[row.stage] * (cycle_spent + cash)
            buy = min(max(0.0, target - cycle_spent), cash)
            shares += buy / row.close
            cash -= buy
            cycle_spent += buy
            max_stage = row.stage
    return _summary("B. 단계별 분할 (20/50/80/100)", df, shares, cash, invested)


def run_single_threshold(df: pd.DataFrame, threshold: float = -10.0) -> dict:
    cash = shares = invested = 0.0
    seen_cycle, bought = -1, False
    for row in df.itertuples():
        if row.is_contrib_day:
            cash += MONTHLY
            invested += MONTHLY
        if row.cycle != seen_cycle:
            seen_cycle, bought = row.cycle, False
        if not bought and row.dd <= threshold + 1e-9 and cash > 0:
            shares += cash / row.close
            cash, bought = 0.0, True
    return _summary(f"C. 단일 임계값 ({threshold:.0f}%)", df, shares, cash, invested)


def _summary(name: str, df: pd.DataFrame, shares: float, cash: float, invested: float) -> dict:
    last = float(df["close"].iloc[-1])
    final = shares * last + cash
    return {
        "strategy": name,
        "invested": round(invested),
        "final_value": round(final),
        "multiple": round(final / invested, 3),
        "end_cash": round(cash),
    }


def main() -> None:
    all_rows = []
    for start in ["2000-01-01", "2010-01-01"]:
        df = load_daily(start)
        for res in [run_regular(df), run_staged(df), run_single_threshold(df)]:
            all_rows.append({"start": start, **res})
    out = pd.DataFrame(all_rows)
    out.to_csv(OUT_CSV, index=False)
    print(out.to_string(index=False))


if __name__ == "__main__":
    main()
```

```python
from __future__ import annotations

import pandas as pd

HIGH_LOOKBACK_DAYS = 252

# 52주 고점 대비 하락률(%) 단계. 경계값은 깊은 단계로 판정하며, 종가 비율 계산의
# 부동소수 오차(-9.999...998 등)가 경계 판정을 뒤집지 않도록 엡실론을 더한다.
_STAGE_THRESHOLDS = [(4, -30.0), (3, -20.0), (2, -10.0), (1, -5.0)]
_EPS = 1e-9


def stage_for_drawdown(drawdown_pct: float) -> int:
    for stage, threshold in _STAGE_THRESHOLDS:
        if drawdown_pct <= threshold + _EPS:
            return stage
    return 0


def compute_daily_state(close: pd.Series) -> pd.DataFrame:
    """일별 52주 고점·하락률·단계·사이클. 사이클은 조정(1단계 이상)을 겪은 뒤
    52주 신고가(하락률 0)를 회복한 날부터 +1."""
    high_52w = close.rolling(HIGH_LOOKBACK_DAYS, min_periods=1).max()
    drawdown_pct = (close / high_52w - 1.0) * 100.0
    stage = drawdown_pct.map(stage_for_drawdown)

    cycle_ids: list[int] = []
    cycle_id = 0
    dipped = False
    for dd, st in zip(drawdown_pct, stage):
        if dipped and dd >= -_EPS:
            cycle_id += 1
            dipped = False
        if st >= 1:
            dipped = True
        cycle_ids.append(cycle_id)

    return pd.DataFrame({
        "close": close,
        "high_52w": high_52w,
        "drawdown_pct": drawdown_pct,
        "stage": stage,
        "cycle_id": cycle_ids,
    })


def extract_signals(daily: pd.DataFrame) -> list[dict]:
    """단계 첫 진입 이벤트. 급락으로 건너뛴 단계는 같은 날 함께 기록한다."""
    signals: list[dict] = []
    max_stage = 0
    current_cycle = -1
    for idx, row in daily.iterrows():
        if int(row["cycle_id"]) != current_cycle:
            current_cycle = int(row["cycle_id"])
            max_stage = 0
        st = int(row["stage"])
        if st > max_stage:
            for s in range(max_stage + 1, st + 1):
                signals.append({
                    "cycle_id": current_cycle,
                    "stage": s,
                    "date": idx.date().isoformat(),
                    "index_close": round(float(row["close"]), 2),
                    "drawdown_pct": round(float(row["drawdown_pct"]), 2),
                })
            max_stage = st
    return signals
```

```python
def add_reference_indicators(daily: pd.DataFrame) -> pd.DataFrame:
    """참고 지표(매수 조건 아님): 주봉 RSI(14), 월봉 20이평 이격도, 200일선 이격도."""
```

```typescript
// S&P 500 조정 매수 전략 상수·계산. 단계 판정 자체는 파이프라인이 하고,
// 여기는 이미 판정된 값으로 "얼마를 살 차례인지"만 계산한다.
export const STAGE_RATIOS: Record<number, number> = { 1: 0.2, 2: 0.3, 3: 0.3, 4: 0.2 }
export const STAGE_LEVELS: Record<number, number> = { 1: -5, 2: -10, 3: -20, 4: -30 }
export const STAGE_LABELS: Record<number, string> = {
  0: '평상', 1: '조정 진입', 2: '조정', 3: '약세장', 4: '폭락',
}

export interface StagePlanRow {
  stage: number
  levelPct: number
  ratio: number
  amount: number
  triggerIndex: number
  triggered: boolean
}

export function buildBuyPlan(
  reserve: number,
  high52w: number,
  triggeredStages: number[],
): StagePlanRow[] {
  return [1, 2, 3, 4].map((stage) => ({
    stage,
    levelPct: STAGE_LEVELS[stage],
    ratio: STAGE_RATIOS[stage],
    amount: Math.round(reserve * STAGE_RATIOS[stage]),
    triggerIndex: high52w * (1 + STAGE_LEVELS[stage] / 100),
    triggered: triggeredStages.includes(stage),
  }))
}

export function nextUntriggeredStage(triggeredStages: number[]): number | null {
  for (const s of [1, 2, 3, 4]) if (!triggeredStages.includes(s)) return s
  return null
}
```

```typescript
import { cacheLife, cacheTag } from 'next/cache'
import { SCREENER_CACHE_TAG } from './queries'
import { createServerSupabaseClient } from './supabase'

export interface Sp500DailyRow {
  date: string
  close: number
  high_52w: number
  drawdown_pct: number
  stage: number
  cycle_id: number
  rsi_w: number | null
  ma20_m_gap: number | null
  ma200_d_gap: number | null
}

export interface Sp500SignalRow {
  cycle_id: number
  stage: number
  date: string
  index_close: number
  drawdown_pct: number
}

export interface Sp500EtfQuoteRow {
  ticker: string
  name: string
  close: number
  change_pct: number
  currency: 'KRW' | 'USD'
  as_of: string
}

const COLS = 'date, close, high_52w, drawdown_pct, stage, cycle_id, rsi_w, ma20_m_gap, ma200_d_gap'
const PAGE = 1000

export async function getSp500Daily(): Promise<Sp500DailyRow[]> {
  'use cache'
  cacheLife('hours')
  cacheTag(SCREENER_CACHE_TAG)
  const supabase = createServerSupabaseClient()
  const rows: Sp500DailyRow[] = []
  for (let from = 0; ; from += PAGE) {
    const { data, error } = await supabase
      .from('sp500_daily')
      .select(COLS)
      .order('date', { ascending: true })
      .range(from, from + PAGE - 1)
    if (error) throw new Error(error.message)
    rows.push(...((data ?? []) as unknown as Sp500DailyRow[]))
    if (!data || data.length < PAGE) break
  }
  return rows
}

export async function getSp500Latest(): Promise<Sp500DailyRow | null> {
  'use cache'
  cacheLife('hours')
  cacheTag(SCREENER_CACHE_TAG)
  const supabase = createServerSupabaseClient()
  const { data, error } = await supabase
    .from('sp500_daily')
    .select(COLS)
    .order('date', { ascending: false })
    .limit(1)
    .maybeSingle()
  if (error) throw new Error(error.message)
  return data as unknown as Sp500DailyRow | null
}

export async function getSp500Signals(): Promise<Sp500SignalRow[]> {
  'use cache'
  cacheLife('hours')
  cacheTag(SCREENER_CACHE_TAG)
  const supabase = createServerSupabaseClient()
  const { data, error } = await supabase
    .from('sp500_signals')
    .select('cycle_id, stage, date, index_close, drawdown_pct')
    .order('date', { ascending: false })
  if (error) throw new Error(error.message)
  return (data ?? []) as Sp500SignalRow[]
}

export async function getSp500EtfQuotes(): Promise<Sp500EtfQuoteRow[]> {
  'use cache'
  cacheLife('hours')
  cacheTag(SCREENER_CACHE_TAG)
  const supabase = createServerSupabaseClient()
  const { data, error } = await supabase
    .from('sp500_etf_quotes')
    .select('ticker, name, close, change_pct, currency, as_of')
    .order('currency', { ascending: false })
  if (error) throw new Error(error.message)
  return (data ?? []) as Sp500EtfQuoteRow[]
}
```

```tsx
'use client'

import { useEffect, useRef, useState } from 'react'
import { createChart, LineStyle } from 'lightweight-charts'
import type { Sp500DailyRow } from '@/lib/sp500'
import { STAGE_LEVELS } from '@/lib/sp500Plan'

type Interval = 'weekly' | 'monthly'

function isoWeekKey(dateStr: string): string {
  const d = new Date(dateStr + 'T00:00:00Z')
  const day = (d.getUTCDay() + 6) % 7
  d.setUTCDate(d.getUTCDate() - day + 3) // 그 주의 목요일 기준 ISO 주차
  const week1 = new Date(Date.UTC(d.getUTCFullYear(), 0, 4))
  const week = 1 + Math.round(
    ((d.getTime() - week1.getTime()) / 86400000 - 3 + ((week1.getUTCDay() + 6) % 7)) / 7,
  )
  return `${d.getUTCFullYear()}-W${String(week).padStart(2, '0')}`
}

function resample(rows: Sp500DailyRow[], interval: Interval) {
  const buckets = new Map<string, Sp500DailyRow>()
  for (const row of rows) {
    const key = interval === 'monthly' ? row.date.slice(0, 7) : isoWeekKey(row.date)
    buckets.set(key, row) // rows는 날짜 오름차순 → 각 구간의 마지막 종가가 남는다
  }
  return [...buckets.values()].map((r) => ({ time: r.date, value: r.close }))
}

const TRIGGER_COLORS: Record<number, string> = {
  1: '#f59e0b', 2: '#f97316', 3: '#ef4444', 4: '#991b1b',
}

export function Sp500Chart({ daily }: { daily: Sp500DailyRow[] }) {
  const containerRef = useRef<HTMLDivElement>(null)
  // 전역 setInterval과의 섀도잉을 피하려고 setIv로 명명
  const [interval, setIv] = useState<Interval>('weekly')

  useEffect(() => {
    if (!containerRef.current || daily.length === 0) return
    const chart = createChart(containerRef.current, {
      height: 380,
      layout: { textColor: '#374151' },
      rightPriceScale: { borderVisible: false },
      timeScale: { borderVisible: false },
    })
    const series = chart.addLineSeries({ color: '#2563eb', lineWidth: 2 })
    series.setData(resample(daily, interval))

    const high52w = daily[daily.length - 1].high_52w
    series.createPriceLine({
      price: high52w, color: '#6b7280', lineWidth: 1,
      lineStyle: LineStyle.Dashed, title: '52주 고점',
    })
    for (const stage of [1, 2, 3, 4]) {
      series.createPriceLine({
        price: high52w * (1 + STAGE_LEVELS[stage] / 100),
        color: TRIGGER_COLORS[stage], lineWidth: 1,
        lineStyle: LineStyle.Dotted, title: `${stage}단계 ${STAGE_LEVELS[stage]}%`,
      })
    }
    chart.timeScale().fitContent()

    const onResize = () => chart.applyOptions({ width: containerRef.current?.clientWidth ?? 600 })
    onResize()
    window.addEventListener('resize', onResize)
    return () => {
      window.removeEventListener('resize', onResize)
      chart.remove()
    }
  }, [daily, interval])

  return (
    <div className="space-y-2">
      <div className="flex gap-1">
        {(['weekly', 'monthly'] as const).map((iv) => (
          <button
            key={iv}
            onClick={() => setIv(iv)}
            className={`rounded-md px-3 py-1 text-xs font-medium ${
              interval === iv ? 'bg-blue-600 text-white' : 'bg-gray-100 text-gray-600'
            }`}
          >
            {iv === 'weekly' ? '주봉' : '월봉'}
          </button>
        ))}
      </div>
      <div ref={containerRef} />
    </div>
  )
}
```
